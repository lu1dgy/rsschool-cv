Code example:

CamelCase Method from CODEWARS: Write simple .camelCase method for strings. All words must have their first letter capitalized without spaces.

String.prototype.camelCase = function() {
    let str = this.split(' ')
    for (let i = 0; i < str.length; i++) {
      if (str[i]) {
        str[i] = str[i][0].toUpperCase() + str[i].slice(1)
      }
    }
    return str.join('')
  }
